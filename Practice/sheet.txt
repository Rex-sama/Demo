class rect{
    private int length;
    private int width;

    public rect(){
        length = 0;
        width = 0;
    }

    public rect(int i){
        length = width = i;
    }

    public rect(int i, int w){
        length = i;
        width = w;
    }

    public int areaRect(){
        return (length * width);
    }
}
